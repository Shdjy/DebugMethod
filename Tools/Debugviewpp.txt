class Program
{
    static void Main()
    {
        Debug.WriteLine("[error]Hello DebugView++ from C#!");
        Debug.WriteLine("Debug 输出: 只在 Debug 模式下有效");
        System.Diagnostics.Trace.WriteLine("Trace 输出: Release 模式下也能看到");

        // 可以添加自定义 TraceListener，比如写入文件
        Trace.Listeners.Add(new TextWriterTraceListener("log.txt"));
        Trace.WriteLine("这条会被写入到 log.txt 文件");
    }
}
